Event Handling

const btn = xjs.withnew('button');
btn.setText('Click me!');
btn.bindEvent('click', function() {
    this.setText('Clicked: ' + new Date().toLocaleTimeString());
});
btn.appendTo(xjs.with('events-result'));

Dynamic Styling

const box = xjs.withnew('div');
box.setStyle({
    width: '100px',
    height: '100px',
    transition: 'all 0.3s'
});

box.bindEvent('mouseenter', function() {
    this.bg().gradient()
        .linear()
        .angle(45)
        .color('#3498db', 0)
        .color('#2ecc71', 100)
        .set();
});

box.bindEvent('mouseleave', function() {
    this.bgColor('#3498db');
});

box.bgColor('#3498db');
box.appendTo(xjs.with('styling-result'));

Animation

const animBox = xjs.withnew('div');
animBox.setStyle({
    width: '50px',
    height: '50px',
    backgroundColor: '#3498db',
    position: 'relative'
});

animBox.bindEvent('click', function() {
    xjs.animate([this], [{
        left: { start: 0, end: 200, ease: 'easeInOutQuad' },
        backgroundColor: { start: '#3498db', end: '#2ecc71' }
    }], 1000);
});

animBox.appendTo(xjs.with('animation-result'));
